   126	function FilterSelect({ value, onChange, options, className = "" }) {
   127	  const [open, setOpen] = useState(false);
   128	  const ref = useRef(null);
   129	  const selectedLabel = options.find((o) => o.value === value)?.label ?? value;
   130	
   131	  useEffect(() => {
   132	    function handleClickOutside(e) {
   133	      if (ref.current && !ref.current.contains(e.target)) setOpen(false);
   134	    }
   135	    document.addEventListener("mousedown", handleClickOutside);
   136	    return () => document.removeEventListener("mousedown", handleClickOutside);
   137	  }, []);
   138	
   139	  return (
   140	    <div ref={ref} className={`relative ${className}`}>
   141	      <button
   142	        type="button"
   143	        onClick={() => setOpen((prev) => !prev)}
   144	        className={`
   145	          flex items-center justify-between gap-3 min-w-[160px] px-4 py-2.5
   146	          rounded-xl border text-sm font-medium text-foreground
   147	          bg-card backdrop-blur-sm
   148	          transition-all duration-300 cursor-pointer select-none
   149	          ${open
   150	            ? "border-cyan-400 shadow-[0_0_12px_rgba(34,211,238,0.45)] ring-1 ring-cyan-400/30"
   151	            : "border-border hover:border-cyan-500/60 hover:shadow-[0_0_8px_rgba(34,211,238,0.25)]"
   152	          }
   153	        `}
   154	      >
   155	        <span>{selectedLabel}</span>
   156	        <ChevronDown
   157	          className={`w-4 h-4 text-cyan-400 transition-transform duration-300 ${open ? "rotate-180" : ""}`}
   158	        />
   159	      </button>
   160	
   161	      <AnimatePresence>
   162	        {open && (
   163	          <Motion.ul
   164	            initial={{ opacity: 0, y: -6, scale: 0.97 }}
   165	            animate={{ opacity: 1, y: 0, scale: 1 }}
   166	            exit={{ opacity: 0, y: -6, scale: 0.97 }}

   169	              absolute z-50 left-0 top-[calc(100%+6px)] min-w-full
   170	              bg-card border border-border
   171	              shadow-[0_0_20px_rgba(34,211,238,0.2)]
   172	              rounded-xl overflow-hidden py-1
   173	            "
   174	          >
   175	            {options.map((opt) => {
   176	              const isSelected = opt.value === value;
   177	              return (
   178	                <li
   179	                  key={opt.value}
   180	                  onClick={() => { onChange(opt.value); setOpen(false); }}

   182	                    flex items-center justify-between gap-3
